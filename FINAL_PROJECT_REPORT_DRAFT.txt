### B.1 FIFO Logic (Python/Django)
```python
def apply_fifo_deduction(medicine_id, quantity_required):
    # Fetch batches sorted by earliest expiry
    batches = Batch.objects.filter(
        medicine_id=medicine_id, 
        quantity__gt=0, 
        expiry_date__gt=timezone.now().date()
    ).order_by('expiry_date')

    for batch in batches:
        if batch.quantity >= quantity_required:
            batch.quantity -= quantity_required
            batch.save()
            return True
        else:
            quantity_required -= batch.quantity
            batch.quantity = 0
            batch.save()
    return quantity_required == 0
```

### B.2 Expiry Status Logic (Python Property)
```python
@property
def expiry_status(self):
    days = (self.expiry_date - timezone.now().date()).days
    if days <= 0:
        return "Expired"
    elif days <= 30:
        return "Critical"
    elif days <= 90:
        return "Warning"
    else:
        return "Safe"
```
